def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)
In[]:
def es_primo(n):
    if n < 2:
        return False
    elif n == 2:
        return True
    else:
        for i in range(2, n):
            if n % i == 0:
                return False
        return True
